// Vercel Serverless Function
export default async function handler(req, res) {
  // POST 요청만 허용합니다.
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'POST 요청만 허용됩니다.' });
  }

  // Vercel 환경 변수에서 구글 API 키를 꺼내옵니다.
  const apiKey = process.env.GEMINI_API_KEY;
  if (!apiKey) {
    return res.status(500).json({ error: 'Vercel 서버에 API 키가 설정되지 않았습니다. (Environment Variables 확인)' });
  }

  // 프론트엔드에서 보낸 데이터를 받습니다.
  const { topic, tone, slideCount } = req.body;

  if (!topic) {
    return res.status(400).json({ error: '주제가 누락되었습니다.' });
  }

  try {
    const layouts = ['cover', 'image_top', 'text_only', 'image_bottom', 'image_top', 'text_only'];
    const layoutSlice = layouts.slice(0, slideCount);
    
    // 프론트엔드에 노출되어 있던 무거운 프롬프트 명령어를 서버로 이동시켜 숨깁니다.
    const prompt = `주제: "${topic}"\n문체: ${tone}\n슬라이드 수: ${slideCount}장\n\n각 슬라이드의 제목과 본문을 작성해주세요. 제목은 2줄 이내, 본문은 핵심만 간결하게 3줄 이내로 작성하세요. 레이아웃은 순서대로 다음을 사용하세요: ${layoutSlice.join(', ')}\n\n반드시 아래와 같은 JSON 형식으로만 응답해주세요. 마크다운 기호 없이 순수 JSON만 출력하세요.\n{\n  "slides": [\n    { "layout": "레이아웃", "title": "제목", "content": "본문" }\n  ]\n}`;

    // 최신 gemini-2.0-flash 모델을 서버 측에서 안전하게 호출합니다.
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;
    
    const payload = {
      contents: [{ parts: [{ text: prompt }] }],
      generationConfig: {
        temperature: 0.8,
        responseMimeType: "application/json"
      }
    };

    const response = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });

    if (!response.ok) {
      const errData = await response.json().catch(() => ({}));
      throw new Error(errData.error?.message || `Google API 오류 (${response.status})`);
    }

    const data = await response.json();
    const textResponse = data.candidates?.[0]?.content?.parts?.[0]?.text;
    
    if (!textResponse) throw new Error("AI 응답이 비어있습니다.");

    // 구글에서 받은 텍스트(JSON 형태)를 파싱해서 프론트엔드로 전달합니다.
    const parsed = JSON.parse(textResponse);
    res.status(200).json(parsed);

  } catch (error) {
    console.error("Vercel Backend Error:", error);
    res.status(500).json({ error: error.message || '서버 내부 처리 중 오류가 발생했습니다.' });
  }
}
